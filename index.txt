async function sendMessage(){
  const apiKey = getApiKey();
  if(!apiKey) {
    alert("Please enter your Groq API Key in 'My Profile' first.");
    switchPanel('profile', document.querySelectorAll('.nav')[3]);
    return;
  }

  const inp=document.getElementById('userInput');
  const text=inp.value.trim();if(!text)return;
  const btn=document.getElementById('sendBtn');btn.disabled=true;
  inp.value='';inp.style.height='auto';
  appendMsg('user',text);
  sessionMessages.push({role:'user',content:text});
  history.push({role:'user',content:`[${mode.toUpperCase()} MODE] ${text}`});
  showTyping();
  try{
    const res=await fetch(API,{method:'POST',
      headers:{'Content-Type':'application/json','Authorization':'Bearer '+apiKey},
      body:JSON.stringify({model:MODEL,max_tokens:900,
        messages:[{role:'system',content:TWIN_PERSONA},...history]})});
    const data=await res.json();
    if(data.error) throw new Error(data.error.message);
    let reply=data.choices?.[0]?.message?.content||'No response.';
    
    // Learning Logic
    let learned=null;
    const lm=reply.match(/\/\/ TWIN LEARNED:\s*(.+)/);
    if(lm){learned=lm[1].trim();reply=reply.replace(/\/\/ TWIN LEARNED:.*/,'').trim();
      learnedThings.push(learned);
      document.getElementById('learnCount').textContent=learnedThings.length+' learned';
      document.getElementById('learnedBadge').textContent=learnedThings.length;
      updateLearnedList();}

    history.push({role:'assistant',content:reply});
    sessionMessages.push({role:'twin',content:reply});
    removeTyping();
    appendMsg('twin',reply,modeMeta[mode]);
    if(learned)appendLearn(learned);
    exchanges++;
    document.getElementById('exchCount').textContent=exchanges+' exchange'+(exchanges!==1?'s':'');
    saveSession();
  }catch(e){removeTyping();appendMsg('twin','Error: ' + e.message);}
  btn.disabled=false;inp.focus();
}

async function runDecision(){
  const apiKey = getApiKey();
  if(!apiKey) { alert("API Key required."); return; }

  const title=document.getElementById('dTitle').value.trim();
  const ctx=document.getElementById('dContext').value.trim();
  const stakes=document.getElementById('dStakes').value;
  if(!title)return;
  const btn=document.getElementById('decideBtn');
  btn.disabled=true;btn.innerHTML='<span class="spinner"></span> Analyzing...';
  document.getElementById('dresult').innerHTML='';
  
  const prompt=`[DECISION MODE] Decision: ${title} Context: ${ctx} Format response as VERDICT: , WHY: , RISKS: , MY MOVE:`;

  try{
    const res=await fetch(API,{method:'POST',
      headers:{'Content-Type':'application/json','Authorization':'Bearer '+apiKey},
      body:JSON.stringify({model:MODEL,max_tokens:800,
        messages:[{role:'system',content:TWIN_PERSONA},{role:'user',content:prompt}]})});
    const data=await res.json();
    const reply=data.choices?.[0]?.message?.content||'';
    // Simplified render for decision
    document.getElementById('dresult').innerHTML=`<div class="dresult">${reply.replace(/\n/g,'<br>')}</div>`;
    exchanges++;document.getElementById('exchCount').textContent=exchanges+' exchanges';
    saveSession();
  }catch(e){document.getElementById('dresult').innerHTML='<div class="dresult" style="color:var(--danger)">Error.</div>';}
  btn.disabled=false;btn.innerHTML='Run Framework →';
}

function saveSession(){
  if(!sessionMessages.length)return;
  const sessions=JSON.parse(localStorage.getItem(STORAGE_KEY)||'[]');
  const idx=sessions.findIndex(s=>s.id===sessionId);
  const obj={id:sessionId,start:sessionStart,exchanges,messages:sessionMessages,learned:learnedThings};
  if(idx>=0)sessions[idx]=obj;else sessions.unshift(obj);
  localStorage.setItem(STORAGE_KEY,JSON.stringify(sessions.slice(0,30)));
}

function renderHistory(){

  if(!sessions.length){body.innerHTML='<div class="empty-state">No saved conversations yet.</div>';return;}
  body.innerHTML=sessions.map((s,i)=>`
    <div class="session-card">
      <div class="sc-head" onclick="toggleSess(${i})">
        <div><div class="sc-title">Session · ${s.start}</div><div class="sc-meta">${s.exchanges} exchanges</div></div>
        <div class="sc-chev" id="chev-${i}">▾</div>
      </div>
      <div class="sc-body" id="sbody-${i}">
        ${s.messages.map(m=>`<div class="hist-msg"><b>${m.role.toUpperCase()}</b>: ${m.content}</div>`).join('')}
      </div>
    </div>`).join('');
}
